#include<iostream>
using namespace std;
int main(){
       int x;
       cin>>x;
       switch(x){
       case 1:
       cout<<"monday";
       break;

       case 2:
       cout<<"tuesday";
       break;

       case 3:
       cout<<"wed";
       break;

       case 4:
       cout<<"thurs";
       break;

       default:
       cout<<"default";
       break;      
}}
